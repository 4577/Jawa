# Creates a new dict with operand and mnemonic swapped for assembling.
_opcodes_by_opname = dict((o[0], (k,) + o[1:]) for k, o in _opcodes.items())


def assemble_opcode(opcode, operands):
    """
    Assembles a single opcode and its operands.
    """
    opcode = opcode.lower()

    t_op, t_operands = _opcodes_by_opname[opcode][:2]
    out = [struct.pack(ubyte, t_op)]
    if t_operands:
        for i, (t_operand_fmt, _) in enumerate(t_operands):
            out.append(struct.pack(t_operand_fmt, operands[i]))

    return ''.join(out)



[docs]def read_instruction(fio, start_pos):
    try:
        op = ubyte.unpack(fio.read(1))[0]
    except struct.error:
        return None

    name, operands = _opcodes[op][:2]

    final_operands = []
    if operands:
        for fmt, type_ in operands:
            final_operands.append(
                Operand(type_, fmt.unpack(fio.read(fmt.size))[0]))

    return Instruction(op, final_operands, start_pos)
